OnMButtonDown / StartAnimation

sub OnInit()
    OnEnter()
end sub

sub OnEnter()
    alpha.value = 100.0
end sub

sub OnLeave()
    alpha.value = 50.0
end sub

sub OnMButtonDown()
    dim dir as Director

    if ContainsMouseCursor() then
        dir = Stage.FindDirector("Default")
        dir.StartAnimation
    end if
end sub
