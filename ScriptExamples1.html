OnEnter/OnLeave

sub OnEnter()
    alpha.value = 100.0
end sub

sub OnLeave()
    alpha.value = 50.0
end sub

sub OnExecPerField()
    dim cube as Container
    cube = Scene.FindContainer("Cube")
    if scaling.x > 1.5 then
        cube.material.diffuse.red = 0.0
    else
        cube.material.diffuse.red = 100.0
    end if
end sub
